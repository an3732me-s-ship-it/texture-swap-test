AFRAME.registerComponent('material-dial', { tick: function () { const stair = document.querySelector('#stairModel'); const dialMarker = document.querySelector('#marker-texture'); if (dialMarker && dialMarker.object3D.visible && stair) { // 1. Get rotation and normalize to 0-360 let rotY = THREE.MathUtils.radToDeg(dialMarker.object3D.rotation.y); if (rotY < 0) rotY += 360; const mesh = stair.getObject3D('mesh'); if (mesh) { mesh.traverse((node) => { if (node.isMesh) { node.material.transparent = true; // 2. The Four-Way Switch if (rotY >= 315 || rotY < 45) { // FRONT: Pink / Concrete node.material.color.set('#DB35CB'); node.material.roughness = 0.8; node.material.metalness = 0.1; node.material.opacity = 0.8; } else if (rotY >= 45 && rotY < 135) { // RIGHT: Blue / Steel node.material.color.set('#355EDB'); node.material.roughness = 0.2; node.material.metalness = 0.9; node.material.opacity = 0.8; } else if (rotY >= 135 && rotY < 225) { // BACK: Cyan / Glass node.material.color.set('#35DBDB'); node.material.roughness = 0.1; node.material.metalness = 0.1; node.material.opacity = 0.4; // More transparent } else if (rotY >= 225 && rotY < 315) { // LEFT: Gold / Wood node.material.color.set('#DBA935'); node.material.roughness = 0.6; node.material.metalness = 0.3; node.material.opacity = 0.8; } } }); } } } });
